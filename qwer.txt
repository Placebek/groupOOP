class Appliance:
	def __init__(self,brand, power):
		self.__brand = brand
		self.__power = power
	
	def get_brand(self):
		return self.__brand
	def get_power(self):
		return self.__power
	def set_brand(self, brand):
		self.__brand = brand
	def get_power(self, power):
		self.__power = power

class WashingMachine(Appliance):
	def __init__(self):
		super().__init__(self,brand, power)
	def wash(self):
		print("Стиральная машина стирает")
	def operate(self):
		print("BRRRR")
	
class Refrigerator(Appliance):
	def __init__(self):
		super().__init__(self,brand, power)
	def cool(self):
		print("Holodilnik ohlozhdaet")
	#@overriding
	def operate(self):
		print("Bzzzz")
