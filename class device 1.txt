class Transport:
    def __init__(self, name, color)
        self.__name = name
        self.__color = color

    def get_name(self):
        return self.__name

    def set_color(self, red):
        self.__color = red

class Car(Transport):
    def sound():
        print('muzyka')

    def ticket():
        print('number')
