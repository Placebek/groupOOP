class Device:
    def __init__(self, name, battery_life):
        self.__name = name
        self.__battery_life = battery_life
    
    def get_name(self):
        return self.__name

    def get_battery_life(self):
        return self.__battery_life
 
    def set_name(self, name):
        self.name = name 

    def set_battery_life(self, battery_life):
        self.battery_life = battery_life
d = Device("iPhone", 100)
print(d.get_name())
d.set_name("Huawei", 97)
print(d.set_name)

class Laptop(Smartphone):
    def __init__(self, name, battery_life, notebook):
        super().__init__(name, battery_life)
        self.notebook = notebook

    def compile_code(self):
        return f"{self.notebook} ноутбук кодты курастырды"

    def use(self):
        return f"{a} = {[self.phone, self.notebook]} "

class Tablet(Laptop):
