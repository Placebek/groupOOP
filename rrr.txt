class ZooAnimal:
    def __init__(self,species,weight):
        self.species=species
	self.weight=weight
    def get_species(self):
	return self.species
    def get_weight(self):
	return self.weight
    def sound(self):
	return:"jivotnoe izdaet zvuk"
class Lion(ZooAnimal):
    def __init__(self,weight):
	super().__init__("lion"weight)
 
    def roar(self)
	return "lev rychit"
    def sound(self)
	return"izdaet zvuk"
class Elephant(ZooAnimal):
    def __init__(self,weight):
	super().__init__("Elephant" weight)

    def trumpet(self)
	return"slon trubit"
    def sound(self)
	return self.trumpet()
class penguin(ZooAnimal):
    def__init__(self,weight):
	super().__init__("penguin" weight)
   
    def swim(self)
	return"penguin plavaet"
    def sound(self)
	return self.swim()

animals=[Lion(250), Elephant(5000), penguin(15)]
for animal in animals:
    print (f"{animals.get_species()} vesit {animal.get_weight()} kg i izdaet zvuk:{animal.sound()}")
